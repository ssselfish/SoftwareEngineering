```python
vowels = "aeiou"
for i in range(1,4):
    print(f"Предложение номер {i}")
    sent = input()
    print(len(sent))
    l_sent = sent.lower()
    print(l_sent)
    count = 0;
    for vowel in sent:
        if vowel in vowels:
            count += 1
    print(count)
    print(sent.replace("ugly", "beauty"))
    print(sent.startswith("The"))
    print(sent.endswith("end"))
```
